create table game(
	game_id int,
	game_name varchar(32),
	primary key (game_id)
);

create table users(	
	user_id int,	
	user_name varchar(32),	
	primary key(user_id)
);

create table player(	
	player_id int,	
	user_id int,	
	game_id int,	
	color int,	
	primary key(player_id),	
	foreign key (user_id) references users(user_id),	
	foreign key (game_id) references game(game_id)
);

create table moves(
	move_id int,
	game_id int,
	fen_string varchar(72),
	capture_piece varchar(72),
	primary key (move_id),
	foreign key (game_id) references game(game_id)
);

create table material(
	mat_id int,
	move_id int,
	mvalue int,
	primary key(mat_id),
	foreign key (move_id) references moves(move_id)
);

create table results(
	result_id int,
	game_id int,
	winner varchar(5),
	primary key (result_id),
	foreign key (game_id) references game(game_id)
);
